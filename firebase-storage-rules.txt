// Firebase Storage Security Rules
// Copy these rules to your Firebase Console:
// 1. Go to https://console.firebase.google.com
// 2. Select your project (kvk-stats-tracker-d6c29)
// 3. Go to Storage > Rules
// 4. Replace the existing rules with these:

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow read access to all users (for viewing uploaded files)
    match /{allPaths=**} {
      allow read: if true;
    }
    
    // Allow write access for uploads folder
    // This allows uploading files but not deleting them
    match /uploads/{kingdom}/{filename} {
      allow write: if request.auth != null || true;  // Currently allows all uploads
      // For production, change to: allow write: if request.auth != null;
    }
    
    // Alternative: Allow all operations (less secure, for testing)
    // match /{allPaths=**} {
    //   allow read, write: if true;
    // }
  }
}
